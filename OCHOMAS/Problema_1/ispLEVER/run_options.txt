#project files

#implementation: "problema 1"
impl -add "problema 1" -type fpga

#device options
set_option -technology ispGAL
set_option -part ispGAL22LV10

set_option -speed_grade -4
set_option -part_companion ""

#compilation/mapping options
set_option -top_module "codigo"

# mapper_options
set_option -frequency 1

set_option -write_vhdl 0
set_option -srs_instrumentation 1

# Lattice ispLSI1K/1KE

# Lattice ispGAL
set_option -areadelay 0

# sequential_optimization_options

# Compiler Options
set_option -compiler_compatible 0

set_option -auto_infer_blackbox 0

#automatic place and route (vendor) options
set_option -write_apr_constraint 1

#set result format/file last
project -result_file "./codigo.edi"

#set log file 
set_option log_file "C:/isplever_classic2_0/examples/ochomas/problema 1/codigo.srf"
